function toCamelCase(str){
  const wordList = str.split(/-|_/)
  let outStr = wordList[0]
  
  wordList.slice(1).forEach((word) => {
    outStr += word[0].toUpperCase() + word.slice(1)
  })
  
  return outStr
}
